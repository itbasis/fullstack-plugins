== use gradle

`buildSrc/build.gradle.kts`:

```gradle
repositories {
    maven(url = "https://dl.bintray.com/borzdeg/ru.itbasis.gradle")
}
dependencies {
	api("ru.itbasis.gradle:spring-boot-backend-all-plugins:0.2")
}
```

in modules (enable any of the plugins):

```gradle
plugins {
    // id("ru.itbasis.root-module") - only root module
    // id("ru.itbasis.spring-boot.backend-library-plugin")
    // id("ru.itbasis.spring-boot.backend-library-db-plugin")
    // id("ru.itbasis.spring-boot.backend-service-plugin")
    // etc...
}
```
